# التحقق من الإصدار

6 التثبيت عبر Docker

6.1 Dockerfile

# Dockerfile
FROM python:3.12-slim

ENV PYTHONDONTWRITEBYTECODE=1
ENV PYTHONUNBUFFERED=1

WORKDIR /app

RUN apt-get update && \
    apt-get install -y --no-install-recommends \
    gcc libpq-dev && \
    rm -rf /var/lib/apt/lists/*

COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

COPY . .

RUN python manage.py collectstatic --noinput

EXPOSE 8000

CMD ["gunicorn", "config.wsgi:application", \
     "--bind", "0.0.0.0:8000", \
     "--workers", "4", \
     "--threads", "2"]

6.2 docker-compose.yml

# docker-compose.yml
version: '3.8'

services:
  db:
    image: postgres:15-alpine
    volumes:
      - postgres_data:/var/lib/postgresql/data
    env_file:
      - .env
    environment:
      POSTGRES_DB: accounting_db
      POSTGRES_USER: db_user
      POSTGRES_PASSWORD: secure_password
    ports:
      - "5432:5432"
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U db_user"]
      interval: 10s
      timeout: 5s
      retries: 5

  redis:
    image: redis:7-alpine
    ports:
      - "6379:6379"
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 10s

  web:
    build: .
    command: >
      sh -c "python manage.py migrate &&
             python manage.py collectstatic --noinput &&
             gunicorn config.wsgi:application
             --bind 0.0.0.0:8000
             --workers 4
             --threads 2"
    volumes:
      - static_volume:/app/staticfiles
      - media_volume:/app/media
    ports:
      - "8000:8000"
    env_file:
      - .env
    depends_on:
      db:
        condition: service_healthy
      redis:
        condition: service_healthy

  nginx:
    image: nginx:alpine
    ports:
      - "80:80"
      - "443:443"
    volumes:
      - static_volume:/app/staticfiles
      - media_volume:/app/media
      - ./nginx/conf.d:/etc/nginx/conf.d
      - ./nginx/ssl:/etc/nginx/ssl
    depends_on:
      - web

volumes:
  postgres_data:
  static_volume:
  media_volume:

7 إعداد PostgreSQL للإنتاج

7.1 إنشاء المستخدم وقاعدة البيانات

# الدخول إلى PostgreSQL كمستخدم postgres
sudo -u postgres psql

# إنشاء المستخدم
CREATE USER db_user WITH PASSWORD 'StrongPassword123!';

# إنشاء قاعدة البيانات
CREATE DATABASE accounting_db OWNER db_user;

# تعيين الترميز
ALTER ROLE db_user SET client_encoding TO 'utf8';

# تعيين عزل المعاملات
ALTER ROLE db_user SET default_transaction_isolation TO 'read committed';

# تعيين المنطقة الزمنية
ALTER ROLE db_user SET timezone TO 'Africa/Cairo';

# منح الصلاحيات
GRANT ALL PRIVILEGES ON DATABASE accounting_db TO db_user;

# الخروج
\q

7.2 صلاحيات الأدوار

# إنشاء دور للقراءة فقط (للتقارير)
CREATE ROLE readonly_role;
GRANT CONNECT ON DATABASE accounting_db TO readonly_role;
GRANT USAGE ON SCHEMA public TO readonly_role;
GRANT SELECT ON ALL TABLES IN SCHEMA public TO readonly_role;
ALTER DEFAULT PRIVILEGES IN SCHEMA public GRANT SELECT ON TABLES TO readonly_role;

# ربط مستخدم للدور
GRANT readonly_role TO report_user;

7.3 إعداد SSL

# تعديل postgresql.conf
# على Ubuntu: /etc/postgresql/15/main/postgresql.conf
ssl = on
ssl_cert_file = '/etc/ssl/certs/ssl-cert-snakeoil.pem'
ssl_key_file = '/etc/ssl/private/ssl-cert-snakeoil.key'
ssl_min_protocol_version = 'TLSv1.2'

# تعديل pg_hba.conf لإلزام SSL
# على Ubuntu: /etc/postgresql/15/main/pg_hba.conf
hostssl accounting_db db_user 0.0.0.0/0 scram-sha-256

# إعادة تشغيل PostgreSQL
sudo systemctl restart postgresql

8 إعداد Nginx كوكيل عكسي

8.1 ملف الإعداد

# /etc/nginx/conf.d/accounting.conf
upstream accounting_app {
    server 127.0.0.1:8000;
}

server {
    listen 80;
    server_name your-domain.com;
    return 301 https://$server_name$request_uri;
}

server {
    listen 443 ssl http2;
    server_name your-domain.com;

    # إعدادات SSL
    ssl_certificate /etc/letsencrypt/live/your-domain.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/your-domain.com/privkey.pem;
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers HIGH:!aNULL:!MD5;
    ssl_prefer_server_ciphers on;

    # تفعيل ضغط
    gzip on;
    gzip_types text/plain text/css application/json application/javascript
               text/xml application/xml text/javascript;

    # الملفات الثابتة
    location /static/ {
        alias /home/user/accounting_system/staticfiles/;
        expires 30d;
        add_header Cache-Control "public, immutable";
    }

    # الملفات المرفوعة
    location /media/ {
        alias /home/user/accounting_system/media/;
        expires 7d;
    }

    # التطبيق الرئيسي
    location / {
        proxy_pass http://accounting_app;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_redirect off;
        client_max_body_size 20M;
    }

    # حماية المجلدات
    location ~ /\. {
        deny all;
    }
}

9 التوزيع عبر CI/CD

9.1 GitHub Actions

# .github/workflows/deploy.yml
name: CI/CD Pipeline

on:
  push:
    branches: [main]
  pull_request:
    branches: [main]

jobs:
  test:
    runs-on: ubuntu-latest

    services:
      postgres:
        image: postgres:15
        env:
          POSTGRES_DB: test_db
          POSTGRES_USER: test_user
          POSTGRES_PASSWORD: test_pass
        ports:
          - 5432:5432
        options: >-
          --health-cmd pg_isready
          --health-interval 10s
          --health-timeout 5s
          --health-retries 5

    steps:
      - uses: actions/checkout@v4

      - name: Set up Python
        uses: actions/setup-python@v5
        with:
          python-version: '3.12'

      - name: Install dependencies
        run: |
          python -m pip install --upgrade pip
          pip install -r requirements.txt

      - name: Run linting
        run: ruff check .

      - name: Run tests
        env:
          DATABASE_URL: postgres://test_user:test_pass@localhost:5432/test_db
          SECRET_KEY: test-secret-key
        run: python manage.py test

  deploy:
    needs: test
    runs-on: ubuntu-latest
    if: github.ref == 'refs/heads/main'

    steps:
      - uses: actions/checkout@v4

      - name: Deploy to server
        uses: appleboy/ssh-action@v1
        with:
          host: ${{ secrets.SERVER_HOST }}
          username: ${{ secrets.SERVER_USER }}
          key: ${{ secrets.SSH_PRIVATE_KEY }}
          script: |
            cd /home/user/accounting_system
            git pull origin main
            source venv/bin/activate
            pip install -r requirements.txt
            python manage.py migrate
            python manage.py collectstatic --noinput
            sudo systemctl restart gunicorn

11 استكشاف أخطاء التثبيت

11.1 مشاكل الاعتماديات

الخطأالحل
ModuleNotFoundError تأكد من تفعيل البيئة الافتراضية وتثبيت requirements.txt
error: Microsoft Visual C++ required ثبّت Visual Studio Build Tools أو استخدم حزم wheels مسبقة البناء
psycopg2 installation error ثبّت libpq-dev (Ubuntu) أو postgresql-devel (RHEL)
pip version conflict شغّل pip install --upgrade pip ثم أعد تثبيت الاعتماديات

11.2 أخطاء قاعدة البيانات

الخطأالحل
connection refused تأكد أن PostgreSQL يعمل: sudo systemctl status postgresql
database does not exist أنشئ قاعدة البيانات يدوياً
permission denied تحقق من صلاحيات المستخدم على قاعدة البيانات
relation already exists شغّل python manage.py migrate --fake-initial أو احذف الجداول المتعارضة
no such column شغّل python manage.py migrate لتطبيق الترحيلات المعلقة

11.3 أخطاء الملفات الثابتة

الخطأالحل
404 for static files شغّل python manage.py collectstatic --noinput
Permission denied on staticfiles غيّر صلاحيات المجلد: chmod -R 755 staticfiles/
ملفات ثابتة قديمة احذف staticfiles/ وأعد التجميع
نصيحة

عند مواجهة أي مشكلة، تحقق أولاً من سجلات الخطأ في logs/django.log أو journalctl -u gunicorn على Linux.
